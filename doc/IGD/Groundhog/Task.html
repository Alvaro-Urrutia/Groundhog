def expand
    if !@dependencies or @dependencies.length == 0 then                
        return [self]
    else
        ret = [self]
        @dependencies.each {|dep|
            ret = ret + dep.expand
        }                   
        return ret
    end
end

def solve(options)
    success = @proc.call(options)
    warn "Fatal: Impossible to solve #{self}" if not success
    return false if not success                
    return success
end
