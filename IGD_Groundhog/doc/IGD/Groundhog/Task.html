def expand
    warn self
    if @dependencies.length == 0 or !@dependencies then                
        return [self]
    else
        ret = [self]
        @dependencies.each {|dep|
            ret = ret + dep.expand
        }                   
        return ret
    end
end

def solvable?
    return true if not @dependencies                
    @dependencies.each{|x|                    
        return false if not x.solvable?
    }
    return true
end

def solve(options)
    success = @proc.call(options)
    warn "Fatal: Impossible to solve #{self}" if not success
    return false if not success                
    return success
end
